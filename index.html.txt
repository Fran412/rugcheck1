let HELIUS_KEY = '';

function saveKey() {
  const val = document.getElementById('heliusKeyInput').value.trim();
  if (!val) { alert('Please paste your Helius API key first.'); return; }
  HELIUS_KEY = val;
  document.getElementById('heliusKeyInput').value = '••••••••••••••••••••••••••••••••';
  document.getElementById('keyStatusBadge').textContent = 'ACTIVE ✓';
  document.getElementById('keyStatusBadge').className = 'key-status set';
  document.getElementById('modeDot').className = 'api-dot live';
  document.getElementById('modeLabel').textContent = 'LIVE MODE — Helius + DexScreener connected.';
}

// ============================================================
// API CALLS
// ============================================================

async function heliusRPC(method, params) {
  const res = await fetch(`https://mainnet.helius-rpc.com/?api-key=${HELIUS_KEY}`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ jsonrpc: '2.0', id: 1, method, params })
  });
  if (!res.ok) throw new Error(`Helius HTTP ${res.status}`);
  const json = await res.json();
  if (json.error) throw new Error(`Helius RPC: ${json.error.message}`);
  return json.result;
}

async function getDexData(address) {
  const res = await fetch(`https://api.dexscreener.com/latest/dex/tokens/${address}`);
  if (!res.ok) throw new Error(`DexScreener HTTP ${res.status}`);
  const json = await res.json();
  const pairs = (json.pairs || []).filter(p => p.chainId === 'solana');
  pairs.sort((a, b) => (b.liquidity?.usd || 0) - (a.liquidity?.usd || 0));
  return pairs[0] || null;
}

// ============================================================
// SCAN
// ============================================================

async function runScan() {
  const addr = document.getElementById('tokenInput').value.trim();
  if (!addr || addr.length < 30) { showError('Enter a valid Solana token contract address (43–44 characters).'); return; }
  if (!HELIUS_KEY) { showError('Please set your Helius API key above first.'); return; }

  hideError();
  document.getElementById('results').classList.remove('visible');
  const fill = document.getElementById('loadingFill');
  const status = document.getElementById('loadingStatus');
  const btn = document.getElementById('scanBtn');
  btn.disabled = true;
  document.getElementById('loadingBar').classList.add('active');
  status.classList.add('active');

  const prog = (p, msg) => { fill.style.width = p + '%'; status.textContent = msg; };

  try {
    prog(10, 'FETCHING DEXSCREENER DATA...');
    let dexData = null;
    try { dexData = await getDexData(addr); } catch(e) { console.warn('DexScreener:', e.message); }

    prog(30, 'READING MINT INFO...');
    let mintInfo = null;
    try {
      const r = await heliusRPC('getAccountInfo', [addr, { encoding: 'jsonParsed' }]);
      mintInfo = r?.value?.data?.parsed?.info || null;
    } catch(e) { console.warn('mintInfo:', e.message); }

    prog(50, 'FETCHING TOKEN METADATA...');
    let tokenMeta = null;
    try { tokenMeta = await heliusRPC('getAsset', { id: addr }); } catch(e) { console.warn('tokenMeta:', e.message); }

    prog(65, 'LOADING HOLDER DISTRIBUTION...');
    let holders = [];
    try {
      const r = await heliusRPC('getTokenLargestAccounts', [addr]);
      holders = r?.value || [];
    } catch(e) { console.warn('holders:', e.message); }

    prog(82, 'SCANNING EARLY TRANSACTIONS...');
    let txns = [];
    try {
      const r = await heliusRPC('getSignaturesForAddress', [addr, { limit: 20 }]);
      txns = r || [];
    } catch(e) { console.warn('txns:', e.message); }

    prog(95, 'CALCULATING VERDICT...');
    await new Promise(r => setTimeout(r, 300));

    const data = analyze(addr, dexData, mintInfo, tokenMeta, holders, txns);
    render(data);

  } catch(err) {
    showError(`Scan error: ${err.message}. Check your API key is valid and the address is a Solana token mint.`);
  } finally {
    document.getElementById('loadingBar').classList.remove('active');
    status.classList.remove('active');

// ANALYSIS — applies book framework
// ============================================================

function analyze(address, dex, mintInfo, tokenMeta, holders, txns) {
  // Token name
  const tokenName = dex?.baseToken?.symbol || tokenMeta?.content?.metadata?.symbol || address.slice(0,8)+'...';

  // Liquidity
  const liquidityUsd = dex?.liquidity?.usd || 0;
  const liquidityLocked = (dex?.labels || []).some(l => String(l).toLowerCase().includes('lock'));

  // Volume & fake volume check (book Ch.4)
  const volume24h = dex?.volume?.h24 || 0;
  const buys24h = dex?.txns?.h24?.buys || 0;
  const sells24h = dex?.txns?.h24?.sells || 0;
  const txns24h = buys24h + sells24h;
  const avgTxSize = txns24h > 0 ? volume24h / txns24h : 0;
  // Flag: avg trade >$30k on a memecoin is wash trading signal
  const fakeVolumeFlag = txns24h > 5 && avgTxSize > 30000;

  // Mint & freeze authority (critical)
  const mintAuthority = mintInfo?.mintAuthority;
  const freezeAuthority = mintInfo?.freezeAuthority;
  const mintRevoked = !mintAuthority;
  const freezeRevoked = !freezeAuthority;
  const supply = mintInfo?.supply ? parseInt(mintInfo.supply) : null;
  const decimals = mintInfo?.decimals ?? 9;
  const supplyReadable = supply ? supply / Math.pow(10, decimals) : null;

  // Holder distribution
  let topHolders = [];
  if (holders.length > 0) {
    const totalRaw = holders.reduce((s, h) => s + parseFloat(h.uiAmount || 0), 0);
    topHolders = holders.slice(0, 10).map((h, i) => {
      const amt = parseFloat(h.uiAmount || 0);
      let pct;
      if (supplyReadable && supplyReadable > 0) {
        pct = parseFloat(((amt / supplyReadable) * 100).toFixed(2));
      } else {
        pct = totalRaw > 0 ? parseFloat(((amt / totalRaw) * 100).toFixed(2)) : 0;
      }
      return { addr: h.address, pct, amt };
    });
  }
  const topHolderPct = topHolders[0]?.pct || 0;
  const holderRisk = topHolderPct > 15 ? 'high' : topHolderPct > 5 ? 'medium' : 'low';

  // Early txn sniper detection (book Ch.3 §5)
  const earlyTxns = txns.slice(0, 20).map((t, i) => ({
    sig: t.signature ? t.signature.slice(0,8)+'...' : `tx${i}`,
    fullSig: t.signature,
    time: t.blockTime ? new Date(t.blockTime*1000).toLocaleTimeString() : '—',
    slot: t.slot,
    err: t.err !== null,
    memo: t.memo
  }));
  const slots = earlyTxns.map(t => t.slot).filter(Boolean);
  const uniqueSlots = new Set(slots).size;
  const sniperFlag = earlyTxns.length >= 5 && uniqueSlots < earlyTxns.length * 0.5;

  // Market data
  const priceUsd = parseFloat(dex?.priceUsd || 0);
  const marketCap = dex?.marketCap || dex?.fdv || 0;
  const priceChange24h = dex?.priceChange?.h24 || 0;
  const pairAge = dex?.pairCreatedAt
    ? Math.floor((Date.now() - dex.pairCreatedAt) / 86400000)
    : null;

  // Score (book framework)
  let score = 100;
  if (!mintRevoked) score -= 25;
  if (!liquidityLocked) score -= 20;
  if (holderRisk === 'high') score -= 20;
  else if (holderRisk === 'medium') score -= 8;
  if (fakeVolumeFlag) score -= 15;
  if (sniperFlag) score -= 10;
  if (!freezeRevoked) score -= 8;
  if (liquidityUsd > 0 && liquidityUsd < 5000) score -= 10;
  if (pairAge !== null && pairAge < 1) score -= 5;
  score = Math.max(5, Math.min(99, score));

  return {
    address, tokenName,
    liquidityUsd, liquidityLocked,
    volume24h, txns24h, buys24h, sells24h, avgTxSize, fakeVolumeFlag,
    mintRevoked, freezeRevoked,
    supplyReadable,
    topHolders, topHolderPct, holderRisk,
    earlyTxns, sniperFlag, uniqueSlots,
    priceUsd, marketCap, priceChange24h, pairAge,
    score,
    dexUrl: dex?.url || null,
    dexId: dex?.dexId || null
  };
}
